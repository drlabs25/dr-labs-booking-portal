// ✅ Fetch report summary function filterReport() { const createdFrom = document.getElementById("createdFrom").value; const createdTo = document.getElementById("createdTo").value; const phlebo = document.getElementById("phleboFilter").value; const agent = document.getElementById("agentFilter").value; const location = document.getElementById("locationFilter").value; fetch(`${API_URL}?action=getReport&createdFrom=${createdFrom}&createdTo=${createdTo}&phleboName=${phlebo}&agentName=${agent}&location=${location}`) .then(res => res.json()) .then(data => { const table = document.querySelector("table"); const thead = table.querySelector("thead"); const body = document.getElementById("reportBody"); body.innerHTML = ""; if (!data || data.length === 0) { body.innerHTML = `No records found`; return; } // --- Build headers dynamically --- let headerRow = ""; if (agent && !phlebo) headerRow += "Agent Name"; if (phlebo && !agent) headerRow += "Phlebo Name"; if (agent && phlebo) { headerRow += "Agent Name"; headerRow += "Phlebo Name"; } if (location) headerRow += "Location"; headerRow += ` No. of Bookings No. of Cancelled No. of Paid Total Confirmed `; thead.innerHTML = headerRow; // --- Fill rows --- data.forEach(row => { let rowHtml = ""; if (agent && !phlebo) rowHtml += `${row.agentName || "-"}`; if (phlebo && !agent) rowHtml += `${row.phleboName || "-"}`; if (agent && phlebo) { rowHtml += `${row.agentName || "-"}`; rowHtml += `${row.phleboName || "-"}`; } if (location) rowHtml += `${row.location || "-"}`; rowHtml += ` ${row.totalBookings} ${row.totalCancelled} ${row.totalPaid} ${row.confirmedBookings} `; body.innerHTML += rowHtml; }); }) .catch(err => alert("Error loading report: " + err)); }
